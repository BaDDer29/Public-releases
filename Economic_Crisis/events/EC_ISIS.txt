# Border war with neighbor country
country_event = {
	id = isis.26
	title = isis.26.t
	desc = isis.26.d
	picture = GFX_event_IGL_army4

	#fire_only_once = yes

	trigger = {
		tag = ISS
		country_exists = ISS
		ISS = {
			has_government = rad_islamists
			is_neighbor_of = FROM 
			any_owned_state = {
				any_neighbor_state = { is_owned_by = FROM }
				ISS = {
					divisions_in_state = {
						state = PREV
						size > 1
					}
				}
			}
		}
		NOT = {
			has_country_flag = border_war_with_ISIS_1
			has_government = rad_islamists
			is_subject_of = ISS
			ISS = { has_war_with = FROM }
		}
		NOT = {
			tag = SYR
			tag = KUR
			tag = IRQ
			tag = FSA
			tag = ROJ
		}
	}

	immediate = {
		hidden_effect = {
			random_owned_state = {
				limit = { any_neighbor_state = { is_owned_by = ISS } }
				set_border_war = yes
			}
			ISS = {
				random_owned_state = {
					limit = { any_neighbor_state = { is_owned_by = FROM } }
					set_border_war = yes
				}
				country_event = { id = isis.27 }
			}
			set_country_flag = border_war_with_ISIS_1
		}
	}

	mean_time_to_happen = {
		days = 12
	}

	option = {
		name = isis.26.a
		log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): isis.26.a"
		add_opinion_modifier = {
			target = ISS
			modifier = attacked_in_our_border
		}
	}
}
